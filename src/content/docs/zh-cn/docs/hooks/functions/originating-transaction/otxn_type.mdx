import { Tabs, TabItem } from '@astrojs/starlight/components';

### 行为

* 返回原始交易的 Transaction Type

### 定义

<Tabs>
<TabItem label="C">
```c
int64_t otxn_type (
    void
);
```
</TabItem>

<TabItem label="JavaScript">
```javascript
function otxn_type(): ErrorCode | number
```
</TabItem>
</Tabs>

### 示例

<Tabs>
<TabItem label="C">
```c
int64_t tt = 
  otxn_type();
```
</TabItem>

<TabItem label="JavaScript">
```javascript
const tt = txn_type()
```
</TabItem>
</Tabs>

### 参数

无

### 返回码

<Tabs>
<TabItem label="C">
| Type     | Description |
| -------- | ----------- |
| int64_t | 原始交易的 Transaction Type。已知类型列表见下表 |
</TabItem>

<TabItem label="JavaScript">
| Type                | Description |
| ------------------- | ----------- |
| ErrorCode \| number | 成功时返回 Transaction Type 数值，失败时返回 ErrorCode。 |
</TabItem>
</Tabs>
