import { Tabs, TabItem, LinkButton } from '@astrojs/starlight/components';

### 概念

<LinkButton href="/docs/hooks/concepts/floating-point-numbers-xfl">浮点数（XFL）</LinkButton>

### 行为

<Tabs>
<TabItem label="C">
* 计算浮点表示的倒数
* 返回新的 XFL
</TabItem>

<TabItem label="JavaScript">
* 将 `1` 除以 XFL
* 返回错误码或新的 XFL bigint
</TabItem>
</Tabs>

### 定义

<Tabs>
<TabItem label="C">
```c
int64_t float_invert (
    int64_t float1
);
```
</TabItem>

<TabItem label="JavaScript">
```javascript
function float_invert(f1: bigint): ErrorCode | bigint
```
</TabItem>
</Tabs>

### 返回码

<Tabs>
<TabItem label="C">
| Type     | Description |
| -------- | ----------- |
| int64_t | <p>XFL（xls17）封装数<br /><br />若为负数，则表示错误：<br /><code>INVALID_FLOAT</code><br />- 参数不是合法 XFL，或结果无法表示<br /><br /><code>DIVISION_BY_ZERO</code><br />- 参数为零。</p> |
</TabItem>

<TabItem label="JavaScript">
| Type                | Description |
| ------------------- | ----------- |
| ErrorCode \| bigint | 返回错误码或倒数 bigint。 |
</TabItem>
</Tabs>
