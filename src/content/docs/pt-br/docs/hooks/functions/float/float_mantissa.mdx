import { Tabs, TabItem, LinkButton, Aside } from '@astrojs/starlight/components';

### Conceitos

<LinkButton href="/pt-br/docs/hooks/concepts/floating-point-numbers-xfl">Números de Ponto Flutuante (XFL)</LinkButton>

### Comportamento

<Tabs>
<TabItem label="C">
* Retorna a parte da mantissa de um XFL como um inteiro sem sinal
</TabItem>

<TabItem label="JavaScript">
* Recupera a mantissa de uma representação de float.
* Um código de erro ou a mantissa como um bigint.
</TabItem>
</Tabs>



### Definição

<Tabs>
<TabItem label="C">
```c
int64_t float_mantissa (
    int64_t float1
);
```
</TabItem>

<TabItem label="JavaScript">
```javascript
function float_mantissa(f1: bigint): ErrorCode | bigint
```
</TabItem>
</Tabs>



### Exemplo

<Tabs>
<TabItem label="C">
```c
int64_t mantissa =
    float_mantissa(float_one());
```

<Aside type="tip">
A mantissa de um XFL negativo é sempre positiva. Use `float_sign` para determinar o sinal do número.
</Aside>
</TabItem>

<TabItem label="JavaScript">
```javascript
const mantissa =
    float_mantissa(float_one());
```
</TabItem>
</Tabs>



### Parâmetros

<Tabs>
<TabItem label="C">
| Nome | Tipo   | Descrição                              |
| ---- | ------ | ---------------------------------------- |
| f1   | bigint | O float do qual recuperar a mantissa. |
</TabItem>

<TabItem label="JavaScript">
| Nome   | Tipo     | Descrição                            |
| ------ | -------- | -------------------------------------- |
| float1 | int64_t | Um número de ponto flutuante de envoltório XFL |
</TabItem>
</Tabs>



### Código de Retorno

<Tabs>
<TabItem label="C">
| Tipo     | Descrição                                                                                                                                                 |
| -------- | ----------------------------------------------------------------------------------------------------------------------------------------------------------- |
| int64_t | <p>A mantissa do XFL<br /><br />Se negativo, um erro:<br /><code>INVALID_FLOAT</code><br />- o parâmetro fornecido não era um número de envoltório XFL válido</p> |
</TabItem>

<TabItem label="JavaScript">
| Tipo                | Descrição                                |
| ------------------- | ------------------------------------------ |
| bigint ou ErrorCode | Um código de erro ou a mantissa como um bigint. |
</TabItem>
</Tabs>
