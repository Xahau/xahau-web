import { Tabs, TabItem, LinkButton } from '@astrojs/starlight/components';

### Conceitos

<LinkButton href="/pt-br/docs/hooks/concepts/floating-point-numbers-xfl">Números de Ponto Flutuante (XFL)</LinkButton>

### Comportamento

<Tabs>
<TabItem label="C">
* Retorna um (`1`) como um XFL int64_t
</TabItem>

<TabItem label="JavaScript">
* Recupera a representação em float de um número.
</TabItem>
</Tabs>



### Definição

<Tabs>
<TabItem label="C">
```c
int64_t float_one();
```
</TabItem>

<TabItem label="JavaScript">
```javascript
function float_one(): ErrorCode | bigint
```
</TabItem>
</Tabs>



### Exemplo

<Tabs>
<TabItem label="C">
```c
int64_t one =
    float_one();
```


</TabItem>

<TabItem label="JavaScript">
```javascript
const one = float_one()
```
</TabItem>
</Tabs>



### Parâmetros

Esta função não possui parâmetros.

### Código de Retorno

<Tabs>
<TabItem label="C">
| Tipo     | Descrição                      |
| -------- | -------------------------------- |
| int64_t | O número de envoltório XFL (xls17) |
</TabItem>

<TabItem label="JavaScript">
| Tipo                | Descrição                      |
| ------------------- | -------------------------------- |
| ErrorCode \| bigint | O número de envoltório XFL (xls17) |
</TabItem>
</Tabs>
