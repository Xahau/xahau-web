import { Tabs, TabItem } from '@astrojs/starlight/components';

### Comportamento

* Retorna o Tipo de Transação da transação de origem

### Definição

<Tabs>
<TabItem label="C">
```c
int64_t otxn_type (
    void
);
```


</TabItem>

<TabItem label="JavaScript">
```javascript
function otxn_type(): ErrorCode | number
```
</TabItem>
</Tabs>



### Exemplo

<Tabs>
<TabItem label="C">
```c
int64_t tt = 
  otxn_type();
```
</TabItem>

<TabItem label="JavaScript">
```javascript
const tt = txn_type()
```
</TabItem>
</Tabs>

### Parâmetros

Nenhum

### Código de Retorno

<Tabs>
<TabItem label="C">
| Tipo     | Descrição                                                                                                                         |
| -------- | ----------------------------------------------------------------------------------------------------------------------------------- |
| int64_t | O Tipo de Transação da transação de origem. Consulte a tabela abaixo para uma lista dos Tipos de Transação conhecidos no momento em que este texto foi escrito |
</TabItem>

<TabItem label="JavaScript">
| Tipo                | Descrição                                                                       |
| ------------------- | --------------------------------------------------------------------------------- |
| ErrorCode \| number | Retorna o Tipo de Transação como um número, ou um ErrorCode caso a recuperação falhe. |
</TabItem>
</Tabs>
