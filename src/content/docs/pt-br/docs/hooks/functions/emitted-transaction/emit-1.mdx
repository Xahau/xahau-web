import { Tabs, TabItem, LinkButton } from '@astrojs/starlight/components';

### Conceitos

<LinkButton href="/pt-br/docs/protocol-reference/transactions/">Transações</LinkButton>

### Comportamento

<Tabs>
<TabItem label="JavaScript">
* Esta função recebe um objeto JSON de transação e o prepara para emissão.
* A transação deve estar completa, exceto pelo campo Account, que deve ser sempre a conta Hook.
</TabItem>
</Tabs>

### Definição

<Tabs>
<TabItem label="JavaScript">
```javascript
function prepare(
    txJson: Record<string, any> | Transaction
  ): ErrorCode | Record<string, any> | Transaction
```
</TabItem>
</Tabs>



### Exemplo

<Tabs>
<TabItem label="JavaScript">
```javascript
const prepared_txn = prepare({
                TransactionType: "Payment",
                Destination: util_raddr(p1address_ns),
                Amount: parseFloat(drops_sent)*2
            })
```
</TabItem>
</Tabs>

<Tabs>

</TabItem>
