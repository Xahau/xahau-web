import { Tabs, TabItem, LinkButton } from '@astrojs/starlight/components';

### 概念

<LinkButton href="/ja/docs/hooks/concepts/emitted-transactions">発行トランザクション</LinkButton>

### 動作

<Tabs>
<TabItem label="C">
* `read_ptr`からトランザクションを読み取る
* トランザクションを発行ルールに照らして検証する
* 有効な場合、トランザクションをコンセンサスに発行する
* 正規のトランザクションハッシュを`write_ptr`に書き込む
</TabItem>

<TabItem label="JavaScript">
* この関数は、指定されたトランザクションJSONを発行します。
* 成功すると、発行されたトランザクションハッシュの数を返します。&#x20;
* エラーがある場合は、エラーコードを返します。
</TabItem>
</Tabs>

### 定義

<Tabs>
<TabItem label="C">
```c
int64_t emit (
    uint32_t write_ptr,
    uint32_t write_len,
    uint32_t read_ptr,
    uint32_t read_len
);
```


</TabItem>

<TabItem label="JavaScript">
```javascript
function emit(
    txJson: Record<string, any> | Transaction
  ): ErrorCode | ByteArray
```
</TabItem>
</Tabs>



### 例

<Tabs>
<TabItem label="C">
```c
if (emit(tx, tx_len) < 0)
    rollback("Failed to emit!", 15, 1);
```


</TabItem>

<TabItem label="JavaScript">
```javascript
const emitResult = emit(txJson)
if(typeof emitResult === 'number')
    rollback("Failed to emit!", 1)
```
</TabItem>
</Tabs>
