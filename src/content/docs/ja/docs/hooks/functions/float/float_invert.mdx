import { Tabs, TabItem, LinkButton } from '@astrojs/starlight/components';

### 概念

<LinkButton href="/docs/hooks/concepts/floating-point-numbers-xfl">Floating Point Numbers (XFL)</LinkButton>

### 動作

<Tabs>
<TabItem label="C">
* XFL に対して `1` を割ります。
* 新しい XFL を int64_t として返します。
</TabItem>

<TabItem label="JavaScript">
* float 表現を反転します。
* エラーコード、または反転後の float を bigint として返します。
</TabItem>
</Tabs>

### 定義

<Tabs>
<TabItem label="C">
```c
int64_t float_invert (
    int64_t float1
);
```
</TabItem>

<TabItem label="JavaScript">
```javascript
function float_invert(f1: bigint): ErrorCode | bigint
```
</TabItem>
</Tabs>

### 例

<Tabs>
<TabItem label="C">
```c
int64_t still_one =
    float_invert(float_one());
```
</TabItem>

<TabItem label="JavaScript">
```javascript
const still_one =
    float_invert(float_one())
```
</TabItem>
</Tabs>

### パラメーター

<Tabs>
<TabItem label="C">
| Name   | Type    | Description                |
| ------ | ------- | -------------------------- |
| float1 | int64_t | XFL 浮動小数点包含数です。 |
</TabItem>

<TabItem label="JavaScript">
| Name | Type   | Description       |
| ---- | ------ | ----------------- |
| f1   | bigint | 反転対象の float。 |
</TabItem>
</Tabs>
